async function lookupBarcode() { const code = document.getElementById('f-barcode').value.trim(); if(!code) return; const btn = document.querySelector('.lookup-btn'); const originalText = btn.innerText; btn.innerText = 'Searching...'; btn.disabled = true; let foundBook = false; try { const url = `https://openlibrary.org/api/volumes/brief/isbn/${code}.json`; const res = await fetch(url); if (res.ok) { const data = await res.json(); if (data && Object.keys(data).length > 0) { const records = data.records; const firstKey = Object.keys(records)[0]; const bookInfo = records[firstKey]?.record; if (bookInfo && bookInfo.title) { foundBook = true; document.getElementById('f-title').value = bookInfo.title; const titleUpper = bookInfo.title.toUpperCase(); const pubSelect = document.getElementById('f-publisher'); if (titleUpper.includes('MARVEL')) { pubSelect.value = 'Marvel'; } else if (titleUpper.includes('DC COMICS') || titleUpper.includes('BATMAN') || titleUpper.includes('SUPERMAN')) { pubSelect.value = 'DC'; } else if (titleUpper.includes('IMAGE COMICS')) { pubSelect.value = 'Image'; } else if (titleUpper.includes('DARK HORSE')) { pubSelect.value = 'Dark Horse'; } else { pubSelect.value = 'Other'; } showToast("Barcode search complete!"); } } } // Handle database blind spots smoothly if (!foundBook) { showToast("Not found in database. Enter title manually."); const titleField = document.getElementById('f-title'); titleField.focus(); titleField.select(); // Ready for immediate typing } } catch (e) { console.error(e); showToast("Network error searching database."); } finally { btn.innerText = originalText; btn.disabled = false; } }
